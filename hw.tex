\begin{homeworkProblem}
\begin{enumerate}[a.]
    \item $(g^{ab}) \; mod \; p$, where g, a, b and p $\in$ N 
    \item  $(g^{ax+by}) \; mod \; p$, where g, a, b, x and y $\in$ N 
    \item  $a|b$, where a and b $\in$ N. 
        Then $\exists$  n $\in$ N, that $$b = ma$$
        Multiply both sides with
        $g \in N$, $$bg=mag$$ Take $n \in N$ where $n=mg$, then $$bg=na$$ a divides bg therefore $a|bg$
\end{enumerate}
\end{homeworkProblem}
